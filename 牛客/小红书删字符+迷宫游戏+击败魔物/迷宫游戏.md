    public static int helper(char[][] board,int n){
        boolean[][] visited=new boolean[n][n];
        int startX=0,startY=0,endX=0,endY=0;
        for(int i=0;i<n;i++){
            for(int j=0;j<n;j++){
                if(board[i][j]=='S'){
                    startX=i;
                    startY=j;
                }
                if(board[i][j]=='E'){
                    endX=i;
                    endY=j;
                }
            }
        }
        dfs(startX,startY,board,n,endX,endY,0,visited);
        
        return dp[endX][endY]==0 ? -1 : dp[endX][endY];
    }
    
    public static void dfs(int i, int j,char[][] board,int n,int endX,int endY,int count,boolean[][] visited){
        if(board[i][j]=='#'){
            return;
        }
        if(visited[i][j]){
            return;
        }
        if(dp[i][j]!=0 && dp[i][j]<count){
            return;
        }
        
        
        
        
        dp[i][j]=count;
        
        if(i==endX && j==endY){
            return;
        }
        visited[i][j]=true;
        dfs(i-1<0 ? n-1:i-1,j,board,n,endX,endY,count+1,visited);
        dfs(i+1==n ? 0 :i+1,j,board,n,endX,endY,count+1,visited);
        dfs(i,j-1<0? n-1:j-1,board,n,endX,endY,count+1,visited);
        dfs(i,j+1==n? 0 :j+1,board,n,endX,endY,count+1,visited);
        visited[i][j]=false;
    }

import java.util.*;

public class Main{
    static int[][] dp;
    static int sx,sy,ex,ey;
    public static void main(String[] args){
        Scanner scanner=new Scanner(System.in);
        int n=Integer.parseInt(scanner.nextLine());
        char[][] board=new char[n][n];
        dp=new int[n][n];
        for(int i = 0;i <n;i++){
            String string = scanner.nextLine();
            for(int j = 0; j < n;j++){
                board[i][j] = string.charAt(j);
                if(board[i][j]=='S'){
                    sx=i;
                    sy=j;
                }
                if(board[i][j]=='E'){
                    ex=i;
                    ey=j;
                }
            }
        }
        scanner.close();
       if(sx==-1 || sy==-1 || ex==-1 ||ey==-1){
           System.out.println(-1);
           return;
       }
       int res=bfs(board,n);
        System.out.println(res);
        
    }
    
    public static int bfs(char[][] board,int n){
        LinkedList<Node> queue=new LinkedList<>();
        queue.offer(new Node(sx,sy,0));
        int x=-1,y=-1; //help gc
        while(!queue.isEmpty()){
            Node node=queue.poll();
            if(node.x==ex && node.y==ey){
                return node.count;
            }
            //上下左右四个方向压入queue
            //上
            x=node.x-1;y=node.y;
            if(x<0) x=n-1;
            if(board[x][y]!='#') queue.offer(new Node(x,y,node.count+1));
            //下
            x=node.x+1;y=node.y;
            if(x==n) x=0;
            if(board[x][y]!='#') queue.offer(new Node(x,y,node.count+1));
            //左
            x=node.x;y=node.y-1;
            if(y<0) y=n-1;
            if(board[x][y]!='#') queue.offer(new Node(x,y,node.count+1));
            //右
            x=node.x;y=node.y+1;
            if(y==n) y=0;
            if(board[x][y]!='#') queue.offer(new Node(x,y,node.count+1));
        }
        return -1;
    }

}
class Node{
    public int x;
    public int y;
    public int count;
    public Node(int x,int y,int count){
        this.x=x;
        this.y=y;
        this.count=count;
    }
}
